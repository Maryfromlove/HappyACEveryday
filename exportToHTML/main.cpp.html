template<const int BCSize = 27> //add Mod as parameter;
struct Binomial_Coefficient {
    ll c[BCSize + 1][BCSize + 1];
    //Pascal's Triangle

    Binomial_Coefficient() {    //add Mod as parameter;
        c[0][0] = 1;
        for (int n = 1; n <= BCSize; ++n) {
            c[n][0] = c[n][n] = 1;
            for (int k = 1; k < n; ++k)
                c[n][k] = (c[n - 1][k - 1] + c[n - 1][k]);
            //c[n][k] = (c[n - 1][k - 1] + c[n - 1][k])%;
        }
    }

    ll operator()(const int &n, const int &m) {
        if (n < m)return -1;//in case.
        return c[n][m];
    }
};

Binomial_Coefficient<> C = Binomial_Coefficient<>();
ll G[33][33];//n,x

ll fpow(ll a, ll b) {
    ll res = 1;
    while (b) {
        if (b & 1)res = res * a;
        a = a * a;
        b >>= 1;
    }
    return res;
}

template<typename tp>
inline void read(tp &x) {
    char ch;
    bool flag = false;
    for (ch = getchar(); !isdigit(ch); ch = getchar())if (ch == '-') flag = true;
    for (x = 0; isdigit(ch); x = x * 10 + ch - '0', ch = getchar());
    x = flag ? -x : x;
}

inline void print(__int128 x) {
    if (x < 0) {
        putchar('-');
        x = -x;
    }
    if (x > 9)
        print(x / 10);
    putchar(x % 10 + '0');
}
