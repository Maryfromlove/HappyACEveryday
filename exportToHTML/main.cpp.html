#include <bits/stdc++.h>
#include <tuple>

#define  debug(x)  cerr <<#x << " = "<<x<<endl
const int MAXN = 1e6 + 59;
const int MOD = 1e9 + 7;
using namespace std;
typedef long long ll;

struct node {
    int a, b, id;

    bool operator<(const node &rht) {
        if (a == rht.a) return b < rht.b;
        return a < rht.a;
    }
};

vector<node> p(MAXN);
vector<int> ans(MAXN);
vector<int> preMax(MAXN);
vector<int> sufMin;

int main() {

    freopen("codecoder.in", "r", stdin);
    freopen("codecoder.out", "w", stdout);
//    p.reserve(MAXN);
//    ans.reserve(MAXN);
//    preMax.resize(MAXN);
//    sufMin.resize(MAXN);

    ios::sync_with_stdio(false);
    cin.tie(0);

    ll n;
    cin >> n;
    p.resize(n);
    ans.resize(n, 0);

    for (auto &pi:p) {
        static int id = 0;
        cin >> pi.a >> pi.b;
        pi.id = id++;
    }

    int n2 = n + 2;
    p.push_back({-MAXN, -MAXN, -1});
    p.push_back({MAXN, MAXN, -1});
    sort(p.begin(), p.end());

    preMax.resize(n2, 0);
    sufMin.resize(n2, MAXN);

    preMax.front() = p.front().b;
    for (int i = 1; i < n2; i++) {
        preMax[i] = max(preMax[i - 1], p[i].b);
    }

    sufMin.back() = p.back().b;
    for (int i = n2 - 2; i >= 0; i--) {
        sufMin[i] = min(sufMin[i + 1], p[i].b);
    }

    for (int i = 1, j = 0; i <= n; ++i) {
        if (preMax[i] < sufMin[i + 1]) {
            while (j != i) {
                ans[p[++j].id] = i - 1;
            }
        }
    }

    for (int i = 0; i < n; i++) {
        cout << ans[i] << endl;
    }

    return 0;
}
